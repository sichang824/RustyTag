8pub fn create_changelog(version: &Version) -> Result<()> {
9    println!("üîÑ Generating CHANGELOG...");
10    let mut file = OpenOptions::new()
11        .write(true)
12        .create(true)
13        .truncate(true)
14        .open("CHANGELOG.md")?;
15
16    // Get current date
17    let date = Local::now().format("%Y-%m-%d").to_string();
18    println!("üìÖ Current date: {}", date);
19
20    // If file is empty, write standard header
21    if file.metadata()?.len() == 0 {
22        writeln!(file, "# Changelog")?;
23        writeln!(file)?;
24        writeln!(file, "## Change Types")?;
25        writeln!(file)?;
26        writeln!(file, "- **Features**: New features or improvements")?;
27        writeln!(file, "- **Bug Fixes**: Bug fixes and patches")?;
28        writeln!(file, "- **Breaking Changes**: Incompatible changes")?;
29        writeln!(file)?;
30        writeln!(file, "## Commit Guidelines")?;
31        writeln!(file)?;
32        writeln!(file, "All notable changes to this project will be documented in this file. See [Conventional Commits](https://www.conventionalcommits.org/) specification when making commits.")?;
33        writeln!(file)?;
34        writeln!(file, "---")?;
35    }
36
37    // Get remote repository URL and previous version
38    let remote_url = crate::utils::git::get_remote_url()?;
39    let previous_version = crate::utils::git::get_latest_tag()?;
40    println!("üîó Remote repository URL: {}", remote_url);
41    println!("üìå Previous version: {}", previous_version);
42
43    // Write version title and comparison link
44    let initial_version =
45        Version::new(semver::Version::new(0, 1, 0)).with_prefix(previous_version.prefix.clone());
46
47    if previous_version.version == initial_version.version {
48        writeln!(
49            file,
50            "### [{}]({}/commits/{}) ({})",
51            version,
52            remote_url.trim_end_matches(".git"),
53            version,
54            date
55        )?;
56    } else {
57        writeln!(
58            file,
59            "### [{}]({}/compare/{}...{}) ({})",
60            version,
61            remote_url.trim_end_matches(".git"),
62            previous_version,
63            version,
64            date
65        )?;
66    }
67    writeln!(file)?;
68
69    // Get commit history
70    println!("üîç Getting commit history...");
71    let commits = if previous_version.version == initial_version.version {
72        println!("‚ö†Ô∏è No previous version found, getting all commits");
73        crate::utils::git::get_git_commits()?
74    } else {
75        println!("üìä Getting new commits after {}", previous_version);
76        crate::utils::git::get_commits_after_tag(&previous_version.to_string())?
77    };
78    println!("‚úÖ Found {} commits", commits.len());
79
80    // Write all commits
81    writeln!(file, "### Commits")?;
82    writeln!(file)?;
83    for commit in &commits {
84        // Skip "chore: release" commits
85        if commit.message.starts_with("chore: release") {
86            continue;
87        }
88        writeln!(
89            file,
90            "* {} ([{}]({}/commit/{}))",
91            commit.message.lines().next().unwrap_or("").trim(),
92            &commit.hash[..7],
93            remote_url.trim_end_matches(".git"),
94            commit.hash
95        )?;
96    }
97    writeln!(file)?;
98
99    println!("‚ú® CHANGELOG.md generated successfully");
100    Ok(())
101}
